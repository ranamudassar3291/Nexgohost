import fetch from "node-fetch";

const CPANEL_USER = "wscreati";
const API_TOKEN = "YOUR_API_TOKEN";

app.post("/install-wordpress", async (req, res) => {
  const { domain, admin_user, admin_pass, admin_email, site_name } = req.body;

  const payload = {
    domain: domain,
    directory: "",
    admin_user: admin_user,
    admin_pass: admin_pass,
    admin_email: admin_email,
    name: site_name || "My WordPress Site"
  };

  console.log("FINAL PAYLOAD:", payload);

  try {
    const response = await fetch("https://yourdomain.com:2083/execute/WordPress/install", {
      method: "POST",
      headers: {
        "Authorization": `cpanel ${CPANEL_USER}:${API_TOKEN}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify(payload)
    });

    const result = await response.json();
    console.log("CPANEL RESPONSE:", result);

    if (result.status === 1) {
      res.json({ success: true });
    } else {
      res.json({ success: false, error: result.errors || result.error });
    }
  } catch (err) {
    res.json({ success: false, error: err.message });
  }
});
