4. Create 20i service file:

Code:

const axios = require('axios');

const API_URL = 'https://api.20i.com/reseller/v1';

async function testConnection(apiKey) {
  try {
    const response = await axios.get(
      `${API_URL}/reseller`,
      {
        headers: {
          Authorization: `Bearer ${apiKey}`,
          'Content-Type': 'application/json'
        }
      }
    );

    return {
      success: true,
      data: response.data
    };

  } catch (error) {
    return {
      success: false,
      error: error.response?.data || error.message
    };
  }
}

module.exports = { testConnection };

5. Create controller:

controllers/20iController.js

const { testConnection } = require('../services/20iService');

exports.test20i = async (req, res) => {
  const { apiKey } = req.body;

  const result = await testConnection(apiKey);

  res.json(result);
};
