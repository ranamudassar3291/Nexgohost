const axios = require('axios');
const https = require('https');

async function installWordPress(req, res) {
    const { 
        server_ip, 
        cpanel_user, 
        cpanel_pass, 
        admin_user, 
        admin_pass, 
        admin_email, 
        domain 
    } = req.body;

    // 1. SSL Bypass (Important for local/unverified servers)
    const agent = new https.Agent({ rejectUnauthorized: false });

    // 2. Softaculous API URL (Port 2083 for cPanel)
    // 'soft=26' is the ID for WordPress
    const apiUrl = `https://${server_ip}:2083/cpsessXXXXXXXXXX/softaculous/index.php?act=software&soft=26&api=json`;

    const formData = new URLSearchParams();
    formData.append('softdomain', domain);
    formData.append('softdirectory', ''); // Root folder pe install karne ke liye
    formData.append('admin_username', admin_user);
    formData.append('admin_pass', admin_pass);
    formData.append('admin_email', admin_email);
    formData.append('autoinstall', '1'); // Yeh sabse zaruri hai

    try {
        const response = await axios.post(apiUrl, formData, {
            auth: {
                username: cpanel_user,
                password: cpanel_pass
            },
            httpsAgent: agent,
            timeout: 120000 // 2 minutes (WordPress takes time to install)
        });

        // 3. Success Detection Logic
        // Softaculous usually returns 'done' or a success message in JSON
        if (response.data && (response.data.done || response.data.status === 'success')) {
            return res.json({ 
                success: true, 
                message: "WordPress Installed Successfully!",
                url: `https://${domain}/wp-admin`
            });
        } else {
            // Agar install nahi hua toh real error pakro
            const errorMsg = response.data.error || "Installation failed: Directory not empty or DB error";
            throw new Error(errorMsg);
        }

    } catch (error) {
        console.error("Installation Blocked:", error.message);
        return res.status(500).json({ 
            success: false, 
            message: `Real Error: ${error.message}` 
        });
    }
}
