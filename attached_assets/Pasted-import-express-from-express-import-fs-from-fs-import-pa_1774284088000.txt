import express from "express";
import fs from "fs";
import path from "path";
import fetch from "node-fetch";
import unzipper from "unzipper";
import mysql from "mysql2/promise";
import { exec } from "child_process";

const app = express();
app.use(express.json());

// ⚙️ VPS CONFIG
const BASE_DIR = "/var/www"; // jahan domains hain
const DB_HOST = "localhost";
const DB_USER = "root";
const DB_PASS = "YOUR_ROOT_PASSWORD";

// 🔥 INSTALL WORDPRESS
app.post("/install-wordpress", async (req, res) => {
  const { domain, admin_user, admin_pass, admin_email } = req.body;

  const sitePath = path.join(BASE_DIR, domain, "public_html");

  const dbName = "wp_" + Date.now();
  const dbUser = "wpuser_" + Date.now();
  const dbPass = Math.random().toString(36).slice(-10);

  try {
    // 1️⃣ Create DB
    const db = await mysql.createConnection({
      host: DB_HOST,
      user: DB_USER,
      password: DB_PASS
    });

    await db.query(`CREATE DATABASE ${dbName}`);
    await db.query(`CREATE USER '${dbUser}'@'localhost' IDENTIFIED BY '${dbPass}'`);
    await db.query(`GRANT ALL PRIVILEGES ON ${dbName}.* TO '${dbUser}'@'localhost'`);
    await db.query(`FLUSH PRIVILEGES`);

    console.log("DB Created:", dbName);

    // 2️⃣ Download WordPress
    const zipPath = path.join(sitePath, "wp.zip");
    const response = await fetch("https://wordpress.org/latest.zip");
    const fileStream = fs.createWriteStream(zipPath);
    await new Promise((resolve) => {
      response.body.pipe(fileStream);
      response.body.on("end", resolve);
    });

    // 3️⃣ Extract
    await fs.createReadStream(zipPath)
      .pipe(unzipper.Extract({ path: sitePath }))
      .promise();

    // Move files from /wordpress folder
    exec(`mv ${sitePath}/wordpress/* ${sitePath}`);

    // 4️⃣ Create wp-config.php
    const configSample = fs.readFileSync(path.join(sitePath, "wp-config-sample.php"), "utf-8");

    const config = configSample
      .replace("database_name_here", dbName)
      .replace("username_here", dbUser)
      .replace("password_here", dbPass);

    fs.writeFileSync(path.join(sitePath, "wp-config.php"), config);

    // 5️⃣ Set permissions
    exec(`chown -R www-data:www-data ${sitePath}`);
    exec(`chmod -R 755 ${sitePath}`);

    // 6️⃣ Final response
    res.json({
      success: true,
      message: "✅ WordPress Installed Successfully",
      login_url: `https://${domain}/wp-admin`,
      db: { dbName, dbUser, dbPass }
    });

  } catch (err) {
    console.error(err);
    res.json({ success: false, error: err.message });
  }
});

app.listen(3000, () => console.log("Server running on port 3000"));
