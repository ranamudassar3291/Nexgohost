Backend: Node.js (Express) to Call cPanel API

Create a file server.js:

import express from "express";
import fetch from "node-fetch";
import cors from "cors";

const app = express();
app.use(cors());
app.use(express.json());

// Endpoint for installing WordPress
app.post("/install-wordpress", async (req, res) => {
  const { domain, directory, admin_user, admin_pass, admin_email, site_name } = req.body;

  try {
    const response = await fetch("https://YOUR_CPANEL_DOMAIN:2083/execute/WordPress/install", {
      method: "POST",
      headers: {
        "Authorization": "cpanel YOUR_CPANEL_USERNAME:YOUR_API_TOKEN",
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        domain: domain,
        directory: directory || "",
        admin_user: admin_user,
        admin_pass: admin_pass,
        admin_email: admin_email,
        language: "en",
        site_name: site_name
      })
    });

    const result = await response.json();

    // Return real install status
    if (result.status === 1) {
      res.json({ success: true, message: "WordPress installed successfully", details: result });
    } else {
      res.json({ success: false, message: result.errors || "Unknown error" });
    }
  } catch (err) {
    res.json({ success: false, message: err.message });
  }
});

app.listen(3000, () => console.log("Server running on port 3000"));

Replace:

YOUR_CPANEL_DOMAIN → your actual cPanel domain.
YOUR_CPANEL_USERNAME → your cPanel username.
YOUR_API_TOKEN → generate from cPanel → API Tokens.
2️⃣ Frontend: Button to Trigger Installation

In your client panel HTML:

<div>
  <input type="text" id="domain" placeholder="Client Domain">
  <input type="text" id="admin_user" placeholder="Admin Username">
  <input type="password" id="admin_pass" placeholder="Admin Password">
  <input type="email" id="admin_email" placeholder="Admin Email">
  <input type="text" id="site_name" placeholder="Site Name">
  <button onclick="installWP()">Install WordPress</button>
</div>

<script>
async function installWP() {
  const data = {
    domain: document.getElementById("domain").value,
    admin_user: document.getElementById("admin_user").value,
    admin_pass: document.getElementById("admin_pass").value,
    admin_email: document.getElementById("admin_email").value,
    site_name: document.getElementById("site_name").value,
    directory: "" // optional subdirectory
  };

  const res = await fetch("http://localhost:3000/install-wordpress", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(data)
  });

  const result = await res.json();
  alert(result.success ? result.message : "Error: " + result.message);
}
</script>

✅ This ensures actual WordPress installs on cPanel, not fake frontend status.
