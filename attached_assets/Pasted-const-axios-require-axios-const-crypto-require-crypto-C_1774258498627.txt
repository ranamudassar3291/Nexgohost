const axios = require('axios');
const crypto = require('crypto');

// CONFIGURATION - cPanel API
const CPANEL_HOST = 'https://your-server:2083';
const CPANEL_USER = 'cpanelusername';
const CPANEL_TOKEN = 'cpanelapitoken';
const CPANEL_API_URL = `${CPANEL_HOST}/json-api/cpanel`;

const DOMAIN = 'example.com';
const WP_PATH = 'public_html'; // folder to install WordPress
const WP_DB_PREFIX = 'wp_';

// Helper function to generate random credentials
function generateCredentials() {
  const username = 'user' + Math.floor(Math.random() * 10000);
  const password = crypto.randomBytes(8).toString('hex');
  return { username, password };
}

// cPanel API call helper
async function cpanelApi(module, func, params = {}) {
  const url = `${CPANEL_API_URL}?cpanel_jsonapi_module=${module}&cpanel_jsonapi_func=${func}&cpanel_jsonapi_version=2`;
  const query = new URLSearchParams(params).toString();
  const headers = {
    Authorization: `cpanel ${CPANEL_USER}:${CPANEL_TOKEN}`,
  };
  const res = await axios.get(`${url}&${query}`, { headers });
  return res.data;
}

// CREATE DATABASE
async function createDatabase(dbName, dbUser, dbPass) {
  await cpanelApi('Mysql', 'create_database', { name: dbName });
  await cpanelApi('Mysql', 'create_user', { name: dbUser, password: dbPass });
  await cpanelApi('Mysql', 'set_privileges_on_database', {
    user: dbUser,
    database: dbName,
    privileges: 'ALL PRIVILEGES',
  });
  console.log('Database created:', dbName);
}

// INSTALL WORDPRESS
async function installWordPress(domain, path) {
  const { username: dbUser, password: dbPass } = generateCredentials();
  const dbName = domain.replace(/\W/g, '_') + '_wp';

  try {
    // 1. Create DB and user
    await createDatabase(dbName, dbUser, dbPass);

    // 2. Download WordPress
    await cpanelApi('Fileman', 'fetch', {
      dir: path,
      url: 'https://wordpress.org/latest.tar.gz',
    });

    // 3. Extract files
    await cpanelApi('Fileman', 'extract', {
      dir: path,
      archive: 'latest.tar.gz',
    });

    // 4. Move WordPress files to root
    await cpanelApi('Fileman', 'mv', {
      sourcefiles: `${path}/wordpress/*`,
      destfiles: `${path}/`,
    });

    // 5. Create wp-config.php dynamically
    const wpConfig = `
define('DB_NAME', '${dbName}');
define('DB_USER', '${dbUser}');
define('DB_PASSWORD', '${dbPass}');
define('DB_HOST', 'localhost');
define('DB_CHARSET', 'utf8');
define('DB_COLLATE', '');
$table_prefix = '${WP_DB_PREFIX}';
define('AUTH_KEY', '${crypto.randomBytes(32).toString('hex')}');
define('SECURE_AUTH_KEY', '${crypto.randomBytes(32).toString('hex')}');
define('LOGGED_IN_KEY', '${crypto.randomBytes(32).toString('hex')}');
define('NONCE_KEY', '${crypto.randomBytes(32).toString('hex')}');
if (!defined('ABSPATH')) define('ABSPATH', __DIR__ . '/');
require_once ABSPATH . 'wp-settings.php';
`;

    await cpanelApi('Fileman', 'save_file', {
      file: `${path}/wp-config.php`,
      data: wpConfig,
    });

    console.log('WordPress installed at', domain);
    return { dbName, dbUser, dbPass, status: 'installed' };
  } catch (err) {
    console.error('WordPress installation failed:', err);
    return { status: 'failed', error: err.message };
  }
}

// CHECK INSTALL STATUS
async function checkWpInstallStatus(domain, path) {
  try {
    const res = await cpanelApi('Fileman', 'fileexists', { file: `${path}/wp-config.php` });
    return res.exists ? 'installed' : 'not_installed';
  } catch (err) {
    return 'not_installed';
  }
}

// EXAMPLE USAGE
(async () => {
  const status = await checkWpInstallStatus(DOMAIN, WP_PATH);
  console.log('Current status:', status);

  if (status === 'not_installed') {
    const installRes = await installWordPress(DOMAIN, WP_PATH);
    console.log('Install result:', installRes);
  } else {
    console.log('WordPress already installed, show login button.');
  }
})();
