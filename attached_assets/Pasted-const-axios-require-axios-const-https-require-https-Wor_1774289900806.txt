const axios = require('axios');
const https = require('https');

// WordPress Install Function
exports.installWordPress = async (req, res) => {
    const { domain, user, pass, email, cpanel_user, cpanel_pass, server_ip } = req.body;

    // Softaculous API URL (Standard cPanel Port 2083)
    // Note: 'soft=26' is the ID for WordPress
    const url = `https://${server_ip}:2083/login/?login_only=1`;

    try {
        // 1. Create a Bypass for Self-Signed SSL
        const agent = new https.Agent({ rejectUnauthorized: false });

        // 2. Setup Softaculous API Parameters
        const params = new URLSearchParams();
        params.append('soft', '26');
        params.append('autoinstall', '1');
        params.append('softdomain', domain);
        params.append('softdirectory', ''); // Empty for root installation
        params.append('admin_username', user);
        params.append('admin_pass', pass);
        params.append('admin_email', email);
        params.append('softdb', 'wp' + Math.floor(Math.random() * 100)); // Auto-generate DB Name

        // 3. Make the API Call to cPanel
        const response = await axios.post(
            `https://${server_ip}:2083/cpsessXXXXXXXXXX/softaculous/index.php?act=software&soft=26&api=json`,
            params,
            {
                auth: {
                    username: cpanel_user,
                    password: cpanel_pass
                },
                httpsAgent: agent,
                timeout: 90000 // 90 Seconds for Installation
            }
        );

        // 4. Detect Success or Failure
        if (response.data && (response.data.done || response.data.includes('congratulations'))) {
            return res.status(200).json({
                status: 'success',
                message: 'WordPress installed successfully!',
                admin_url: `https://${domain}/wp-admin`
            });
        } else {
            // If Softaculous returns error in JSON
            throw new Error(response.data.error || 'Softaculous rejected the installation.');
        }

    } catch (error) {
        console.error('Installation Error:', error.message);
        
        // Return Real Error to UI
        res.status(500).json({
            status: 'failed',
            message: error.response ? `Server Error: ${error.response.status}` : error.message
        });
    }
};
